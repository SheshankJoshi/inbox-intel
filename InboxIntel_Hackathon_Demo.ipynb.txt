    "    def _extract_message_content(self, response_json: dict[str, Any]) -> str:\n",
    "        choices = response_json.get('choices', [])\n",
    "        if not choices:\n",
    "            raise LLMResponseError('LLM response has no choices')\n",
    "\n",
    "        message = choices[0].get('message', {})\n",
    "        content = message.get('content')\n",
    "        if isinstance(content, str):\n",
    "            return content.strip()\n",
    "\n",
    "        if isinstance(content, list):\n",
    "            text_parts: list[str] = []\n",
    "            for part in content:\n",
    "                if isinstance(part, dict) and part.get('type') == 'text':\n",
    "                    text_parts.append(str(part.get('text', '')))\n",
    "            joined = ''.join(text_parts).strip()\n",
    "            if joined:\n",
    "                return joined\n",
    "\n",
    "        raise LLMResponseError('Unable to extract text content from LLM response')\n",

    "# Tools: email readers and article metadata fetcher\n",
    "\n",
    "class BaseEmailReaderMCP(ABC):\n",
    "    @abstractmethod\n",
    "    def fetch_unread(self, limit: int = 25) -> list[EmailMessage]:\n",
    "        raise NotImplementedError\n",
    "\n",
    "\n",
    "class SampleEmailReaderMCP(BaseEmailReaderMCP):\n",
    "    def __init__(self, sample_dir: str = 'demo/sample_emails') -> None:\n",
    "        self.sample_dir = Path(sample_dir)\n",
    "\n",
    "    def fetch_unread(self, limit: int = 25) -> list[EmailMessage]:\n",
    "        emails: list[EmailMessage] = []\n",
    "        json_files = sorted(self.sample_dir.glob('*.json'))[:limit]\n",
    "        for path in json_files:\n",
    "            with path.open('r', encoding='utf-8') as fh:\n",
    "                payload = json.load(fh)\n",
    "            emails.append(\n",
    "                EmailMessage(\n",
    "                    sender=payload.get('sender', 'unknown@example.com'),\n",
    "                    subject=payload.get('subject', '(no subject)'),\n",
    "                    timestamp=datetime.fromisoformat(payload.get('timestamp', datetime.now(timezone.utc).isoformat())),\n",
    "                    html=payload.get('html', ''),\n",
    "                    text=payload.get('text', ''),\n",
    "                    source_id=payload.get('id', path.stem),\n",
    "                )\n",
    "            )\n",
    "        return emails\n",
    "\n",
    "\n",
    "class ImapEmailReaderMCP(BaseEmailReaderMCP):\n",
    "    def __init__(self) -> None:\n",
    "        self.host = os.getenv('INBOX_INTEL_IMAP_HOST', '')\n",
    "        self.port = int(os.getenv('INBOX_INTEL_IMAP_PORT', '993'))\n",
    "        self.username = os.getenv('INBOX_INTEL_EMAIL_USERNAME', '')\n",
    "        self.password = os.getenv('INBOX_INTEL_EMAIL_PASSWORD', '')\n",
    "        self.mailbox = os.getenv('INBOX_INTEL_IMAP_MAILBOX', 'INBOX')\n",
    "\n",
    "    def fetch_unread(self, limit: int = 25) -> list[EmailMessage]:\n",
    "        if not all([self.host, self.username, self.password]):\n",
    "            raise ValueError('IMAP credentials are missing. Configure environment variables first.')\n",
    "\n",
    "        conn = imaplib.IMAP4_SSL(self.host, self.port)\n",
    "        try:\n",
    "            conn.login(self.username, self.password)\n",
    "            conn.select(self.mailbox)\n",
    "            status, data = conn.search(None, 'UNSEEN')\n",
    "            if status != 'OK':\n",
    "                return []\n",
    "\n",
    "            ids = (data[0] or b'').split()\n",
    "            chosen_ids = ids[-limit:]\n",
    "            emails: list[EmailMessage] = []\n",
    "            for email_id in chosen_ids:\n",
    "                message_id = email_id.decode('utf-8', errors='ignore')\n",
    "                status, msg_data = conn.fetch(message_id, '(RFC822)')\n",
    "                if status != 'OK' or not msg_data or msg_data[0] is None:\n",
    "                    continue\n",
    "\n",
    "                row = msg_data[0]\n",
    "                raw = row[1] if isinstance(row, tuple) and len(row) >= 2 else None\n",
    "                if isinstance(raw, bytearray):\n",
    "                    raw = bytes(raw)\n",
    "                if raw is None:\n",
    "                    continue\n",
    "\n",
    "                msg = message_from_bytes(raw)\n",
    "                emails.append(self._parse_email(msg, source_id=message_id))\n",
    "            return emails\n",
    "        finally:\n",
    "            try:\n",
    "                conn.close()\n",
    "            except Exception:\n",
    "                pass\n",
    "            conn.logout()\n",
    "\n",
    "    def _parse_email(self, msg: Message, source_id: str) -> EmailMessage:\n",
    "        sender = msg.get('From', 'unknown@example.com')\n",
    "        subject = msg.get('Subject', '(no subject)')\n",
    "\n",
    "        date_header = msg.get('Date')\n",
    "        if date_header:\n",
    "            try:\n",
    "                timestamp = parsedate_to_datetime(date_header)\n",
    "            except Exception:\n",
    "                timestamp = datetime.now(timezone.utc)\n",
    "        else:\n",
    "            timestamp = datetime.now(timezone.utc)\n",
    "\n",
    "        html_content = ''\n",
    "        text_content = ''\n",
    "\n",
    "        if msg.is_multipart():\n",
    "            for part in msg.walk():\n",
    "                content_type = part.get_content_type()\n",
    "                payload = part.get_payload(decode=True)\n",
    "                if payload is None:\n",
    "                    continue\n",
    "                charset = part.get_content_charset() or 'utf-8'\n",
    "                if not isinstance(payload, (bytes, bytearray)):\n",
    "                    continue\n",
    "                decoded = payload.decode(charset, errors='ignore')\n",
    "                if content_type == 'text/html' and not html_content:\n",
    "                    html_content = decoded\n",
    "                elif content_type == 'text/plain' and not text_content:\n",
